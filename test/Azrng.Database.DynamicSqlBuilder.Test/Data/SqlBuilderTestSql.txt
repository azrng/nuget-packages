create table public.inventory_details
(
    id                 bigint generated by default as identity,
    order_id           bigint                                    not null,
    order_name         text                                      not null,
    order_item_id      bigint                                    not null,
    product_id         bigint                                    not null,
    vendor_id          bigint                                    not null,
    sales_agent_id     bigint                                    not null,
    product_batch_id   bigint                                    not null,
    batch_no           text                                      not null,
    quantity           numeric(18, 2)                            not null,
    order_details_type integer                                   not null,
    cost_price         numeric(18, 2)                            not null,
    outbound_price     numeric(18, 2),
    total_amount       numeric(18, 2)                            not null,
    creator            varchar(50) default ''::character varying not null,
    create_time        timestamp                                 not null,
    updater            varchar(50) default ''::character varying not null,
    update_time        timestamp,
    deleted            boolean                                   not null,
    disabled           boolean                                   not null
);
