create table aa."user2"
(
    id          bigint,
    account     varchar(20)                                not null,
    password    varchar(200)                               not null,
    nick_name   varchar(10)                                not null,
    avatar      varchar(200) default ''::character varying not null,
    gender      integer      default 0                     not null,
    open_id     varchar(50)  default ''::character varying not null,
    union_id    varchar(50)  default ''::character varying not null,
    remark      varchar(200) default ''::character varying not null,
    email       varchar(100) default ''::character varying not null,
    creater     text,
    create_time timestamp                                  not null,
    modifyer    text,
    modify_time timestamp                                  not null,
    deleted     boolean                                    not null,
    disabled    boolean                                    not null
);
